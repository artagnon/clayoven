def initialize(filename, git)
  super
  # There cannot be ContentPages nested under 'index'
  @topic, @subtopic, = @filename.split "/", 3
  @subtopic = nil if @subtopic.end_with?(".clay")
  @permalink = @filename.split(".clay").first
  @target = "#{@permalink}.html"
end
