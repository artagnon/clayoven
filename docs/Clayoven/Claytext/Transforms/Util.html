def self.to_arabic(str)
  result = 0
  ROMAN_NUMERALS.each_value do |roman|
    while str.start_with?(roman)
      result += ROMAN_NUMERALS.invert[roman]
      str = str.slice(roman.length, str.length)
    end
  end
  result
end
